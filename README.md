# Algorithm MaxOfTwo
Input: Two numbers A and B
Output: The greater number

If A > B then
    Return A
Else
    Return B
EndIf

# Algorithm SumN
Input: Integer N
Output: Sum of numbers from 1 to N

Set sum = 0
For i = 1 to N do
    sum = sum + i
EndFor
Return sum


# Algorithm EvenOdd
Input: Number N
Output: "Even" or "Odd"

If N mod 2 == 0 then
    Return "Even"
Else
    Return "Odd"
EndIf
